Production-Ready Color Lerp

function lerpHue(h1, h2, t) {
    // Convert hues to radians
    const theta1 = h1 * Math.PI / 180;
    const theta2 = h2 * Math.PI / 180;
    
    // Embed on unit circle
    const v1 = [Math.cos(theta1), Math.sin(theta1)];
    const v2 = [Math.cos(theta2), Math.sin(theta2)];
    
    // Interpolate vectors
    const vt = [
        (1 - t) * v1[0] + t * v2[0],
        (1 - t) * v1[1] + t * v2[1]
    ];
    
    // Normalize and convert back
    const norm = Math.sqrt(vt[0]**2 + vt[1]**2);
    const thetaT = Math.atan2(vt[1]/norm, vt[0]/norm);
    
    return (thetaT * 180 / Math.PI + 360) % 360;
}

// CSS gradient generator
function generateGradient(h1, h2, steps) {
    const colors = [];
    for (let i = 0; i <= steps; i++) {
        const t = i / steps;
        const h = lerpHue(h1, h2, t);
        colors.push(`hsl(${h}, 100%, 50%)`);
    }
    return `linear-gradient(90deg, ${colors.join(', ')})`;
}
