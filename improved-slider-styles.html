function circularAverage(a1, a2) {
  // Convert to radians
  const theta1 = a1 * Math.PI / 180;
  const theta2 = a2 * Math.PI / 180;
  
  // Embed on unit circle
  const v1 = [
    Math.cos(theta1), 
    Math.sin(theta1)
  ];
  const v2 = [
    Math.cos(theta2), 
    Math.sin(theta2)
  ];
  
  // Average vectors
  const avgVec = [
    (v1[0] + v2[0])/2, 
    (v1[1] + v2[1])/2
  ];
  
  // Normalize
  const norm = Math.sqrt(
    avgVec[0]**2 + avgVec[1]**2
  );
  avgVec[0] /= norm;
  avgVec[1] /= norm;
  
  // Convert back to angle
  const avgTheta = Math.atan2(
    avgVec[1], avgVec[0]
  );
  return (avgTheta * 180/Math.PI + 360) % 360;
}
